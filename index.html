`; const kpiHtml = `

${pM(tb)}

${pM(tg)}
%${pF(tb>0?tg/tb*100:0,1)} kullanım
Fiziksel
%${fiz}
CPI
${pF(cpi)}
EAC: ${pM(eac)}
`; const pypRows = pyp.map(p=>`${p.kod}${p.ad}A${p.bas}A${p.bit}${pM(p.butce)}${pM(p.g)}
${p.il||0}%
${pDA(p.il||0)}`).join(''); const pypTableHtml = `

TOPLAM${pM(tb)}${pM(tg)}

EVM Sapma Analizi
${evmRows}
PYPİş KalemiEVACCVCV%Risk
